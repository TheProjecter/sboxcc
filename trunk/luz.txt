#include <avr/io.h>
#include <util/delay.h>
#include "easybutterfly.h"
#include <AVRBFLY.h>
PROGRAM_INIT
 
VAR(i)
PROGRAM_START


	if(!(PINB & (1 << 1)) )  {  //port in
		while(!(PINB & (1 << 1))){
		}
		i=i+1;
		
		if(i==1){
		//temp
			LCD_TEXT("WELCOME!")
		//	LCD_puts("WELCOME!");
			PORTD = PORTD & ~(1 << 0); //port out
    	//	SET_BIT(PORTD,0) //led on (bit 0 of portD to 1)
		} //end if 

		else{
			LCD_TEXT("ALARM")
		//	LCD_puts("ALARM");
			PORTD = PORTD & ~(1 << 0);
		//	CLEAR_BIT(PORTD,0) //led off (bit 0 of portD to 0)
		} //end else

	
	}//end if
	


PROGRAM_END
